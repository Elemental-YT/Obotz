#include<avr/io.h>
#include<util/delay.h>
int main(void)
{
  int period = 500;   // Variable to ensure delay times are the same
  DDRD = 0xF0;    // We're only using the Stepper so we only use D4, D5, D6, and D7
  while(1)
  {
    for(int i=0;i<12;i++)   // Repeat this 12 times   (4 Steps X 12 repeats = 48 steps = 360 degrees)
    {
      PORTD = 0b11100000;   // Move CCW by 1 step
      _delay_ms(period);
      PORTD = 0b11010000;   // Move CCW by 1 step
      _delay_ms(period);
      PORTD = 0b10110000;   // Move CCW by 1 step
      _delay_ms(period);
      PORTD = 0b01110000;   // Move CCW by 1 step
      _delay_ms(period);
    }

    for(int i=0;i<12;i++)     // Repeat this 12 times   (4 Steps X 12 repeats = 48 steps = 360 degrees
    {
      PORTD = 0b01110000;     // Move CW by 1 step
      _delay_ms(period);
      PORTD = 0b10110000;     // Move CW by 1 step
      _delay_ms(period);
      PORTD = 0b11010000;     // Move CW by 1 step
      _delay_ms(period);
      PORTD = 0b11100000;     // Move CW by 1 step
      _delay_ms(period);
    }
  }
}

// D4 = Yellow Wire
   D5 = Brown Wire
   D6 = Orange Wire
   D7 = Black Wire
